# 572. Subtree of Another Tree
> https://leetcode.com/problems/subtree-of-another-tree/description/

- `Trees`
- t: O(n * m)
- s: O(n + m)

```py
from collections import deque

class Solution:
    @staticmethod
    def is_same(root1: TreeNode, root2: TreeNode) -> bool:    
        queue = deque([(root1, root2)])
        while queue:
            n1, n2 = queue.popleft()
            if not (n1 and n2):
                if n1 != n2:
                    return False
            elif n1.val != n2.val:
                return False
            else:
                queue.append((n1.left, n2.left))
                queue.append((n1.right, n2.right))
        return True

    def isSubtree(self, root: Optional[TreeNode], subRoot: Optional[TreeNode]) -> bool:
        queue = deque([root,])
        while queue:
            node = queue.popleft()
            if node.val == subRoot.val and Solution.is_same(node, subRoot):
                return True
            queue.extend([c for c in (node.left, node.right) if c])
        return False
```

```cpp
class Solution {
    bool isSame(TreeNode* n1, TreeNode* n2) {
        if (!n1 && !n2) {
            return true;
        } else if (!(n1 && n2) || n1->val != n2->val){
            return false;
        }
        return isSame(n1->left, n2->left) && isSame(n1->right, n2->right);
    }

public:
    bool isSubtree(TreeNode* root, TreeNode* subRoot) {
        if (!root) return false;
        if (root->val == subRoot->val) {
            if (isSame(root, subRoot)) return true;
        }
        return isSubtree(root->left, subRoot) || isSubtree(root->right, subRoot);
    }
};
```
