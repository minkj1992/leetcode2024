# 297. Serialize and Deserialize Binary Tree
> https://leetcode.com/problems/serialize-and-deserialize-binary-tree/description/

- `Tree`
- preorder + recursive

```py
NULL = -1001

class Codec:

    def serialize(self, root):
        nodes = []
        def dfs(node):
            if not node:
                nodes.append(NULL)
                return
            nodes.append(node.val)
            dfs(node.left)
            dfs(node.right)
        
        dfs(root)
        return ','.join(map(str, nodes))
        

    def deserialize(self, data):
        i, nodes = 0, list(map(int, data.split(',')))
        
        def dfs():
            nonlocal i
            val = nodes[i]
            print(val)
            i += 1
            if val == NULL:
                return None
            
            root = TreeNode(val)
            root.left = dfs()
            root.right = dfs()
            return root

        return dfs()
```
