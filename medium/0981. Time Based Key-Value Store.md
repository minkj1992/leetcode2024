# 981. Time Based Key-Value Store
> https://leetcode.com/problems/time-based-key-value-store/description/

- `Binary Search`
- Binary search에서, res를 관리하면 복잡한 코드를 좀 더 쉽게 할 수 있다.

```py
from collections import defaultdict

class TimeMap:

    def __init__(self):
        self.kv = defaultdict(list) # Pair: (value, timestamp)

    def set(self, key: str, value: str, timestamp: int) -> None:
        self.kv[key].append((value,timestamp))
        

    def get(self, key: str, timestamp: int) -> str:
        res = ""
        values = self.kv[key]
        
        l, r = 0, len(values) - 1
        while l <= r:
            mid = (l + r) // 2
            if values[mid][1] <= timestamp:
                res = values[mid][0]
                l = mid + 1
            else:
                r = mid -1
        return res
```

```cpp
class TimeMap {
  unordered_map<string, vector<pair<string, int>>> kv;

 public:
  TimeMap() {}

  void set(const string& key, const string& value, int timestamp) {
    kv[key].emplace_back(value, timestamp);
  }

  string get(const string& key, int timestamp) {
    if (kv.find(key) == kv.end()) return "";
    
    string res = "";
    const auto& vals = kv[key];
    int l = 0, r = vals.size() - 1;
    
    while (l <= r) {
      int mid = l + (r - l) / 2;
      if (vals[mid].second <= timestamp) {
        res = vals[mid].first;
        l = mid + 1;
      } else {
        r = mid - 1;
      }
    }
    return res;
  }
};
```

> Find left most equal or right mos smaller

```py
"""
Left most Equal or Right most smaller (return r)
Return -1 case
    1. n == 0
    2. Every elements are bigger than timestamp
"""
from collections import defaultdict

class TimeMap:
    def __init__(self):
        self.kv = defaultdict(list) # Pair: (timestamp, value)

    def bisect(self, key: str, timestamp: int) -> int:
        l, r = 0, len(self.kv[key])-1
        while l <= r:
            mid = (l+r) // 2
            if timestamp == self.kv[key][mid][0]:
                return mid
            elif timestamp < self.kv[key][mid][0]:
                r = mid - 1
            else:
                l = mid + 1
        return r

    def set(self, key: str, value: str, timestamp: int) -> None:
        self.kv[key].append((timestamp, value))            
        
    def get(self, key: str, timestamp: int) -> str:
        idx = self.bisect(key, timestamp)
        if idx == -1:
            return ""
        return self.kv[key][idx][1]
```
